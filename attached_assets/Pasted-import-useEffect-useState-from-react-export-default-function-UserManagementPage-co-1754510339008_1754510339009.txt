import { useEffect, useState } from "react";

export default function UserManagementPage() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState("");

  useEffect(() => {
    fetch("/api/admin/users") // você pode ter uma rota para listar usuários
      .then(res => res.json())
      .then(data => setUsers(data))
      .catch(err => setError("Erro ao carregar usuários"))
      .finally(() => setLoading(false));
  }, []);

  const updateUserType = async (userId: string, userType: string) => {
    try {
      const res = await fetch(`/api/admin/user-type/${userId}`, {
        method: "PUT",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ userType }),
      });

      const data = await res.json();
      alert(data.message);
    } catch (err) {
      alert("Erro ao atualizar tipo de usuário");
    }
  };

  if (loading) return <p>Carregando...</p>;
  if (error) return <p>{error}</p>;

  return (
    <div className="p-4">
      <h2 className="text-xl font-bold mb-4">Gerenciar Usuários</h2>
      <table className="w-full border">
        <thead>
          <tr>
            <th>Email</th>
            <th>Tipo atual</th>
            <th>Alterar para</th>
          </tr>
        </thead>
        <tbody>
          {users.map((user: any) => (
            <tr key={user.id}>
              <td>{user.email}</td>
              <td>{user.userType || "nenhum"}</td>
              <td>
                <select
                  defaultValue={user.userType}
                  onChange={(e) => updateUserType(user.id, e.target.value)}
                >
                  <option value="">Selecione</option>
                  <option value="solicitante">Solicitante</option>
                  <option value="fornecedor">Fornecedor</option>
                  <option value="visualizador">Visualizador</option>
                </select>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}
